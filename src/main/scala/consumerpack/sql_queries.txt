5. change in price/cost ("sale") for products by timeframe

-- with extract()
select price as sale, extract (year_month from datetime) from TABLE_NAME group by price asc

-- with REGEXP()
select price as sale from TABLE_NAME where datetime REGEXP '%[1-2][0-9][0-9][0-9]-[0-1][0-9]%' group by price asc;

-- with rlike()
select price as sale from TABLE_NAME where rlike(datetime, %[1-2][0-9][0-9][0-9]-[0-1][0-9]%) group by price asc;


6. popular cities/countries by most purchases

select city, sum(payment_txn_id) from TABLE_NAME where payment_txn_success = 'Y' group by city
select country, sum(payment_txn_id) from TABLE_NAME where payment_txn_success = 'Y' group by country
